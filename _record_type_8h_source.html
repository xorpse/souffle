37 /**
38  * @class RecordType
39  * @brief Record type class for record type declarations
40  *
41  * A record type aggregates a list of fields (i.e. name & type) into a new type.
42  * Each record type has a name making the record type unique.
43  */
44 class RecordType : public Type {
45 public:
46  RecordType(QualifiedName name, VecOwn<Attribute> fields, SrcLocation loc = {})
47  : Type(std::move(name), std::move(loc)), fields(std::move(fields)) {}
48 
49  /** Add field to record type */
50  void add(std::string name, QualifiedName type) {
51  fields.push_back(mk<Attribute>(std::move(name), std::move(type)));
52  }
53 
54  /** Get fields of record */
55  std::vector<Attribute*> getFields() const {
56  return toPtrVector(fields);
57  }
58 
59  /** Set field type */
60  void setFieldType(size_t idx, QualifiedName type) {
61  fields.at(idx)->setTypeName(std::move(type));
62  }
63 
64  RecordType* clone() const override {
65  return new RecordType(getQualifiedName(), souffle::clone(fields), getSrcLoc());
66  }
67 
68 protected:
69  void print(std::ostream& os) const override {
70  os << tfm::format(".type %s = [%s]", getQualifiedName(), join(fields, ", "));
71  }
72 
73  bool equal(const Node& node) const override {
74  const auto& other = dynamic_cast<const RecordType&>(node);
75  return getQualifiedName() == other.getQualifiedName() && equal_targets(fields, other.fields);
76  }
77 
78 private:
79  /** record fields */
80  VecOwn<Attribute> fields;
81 };
